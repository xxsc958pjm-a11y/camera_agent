function parseParameters(data) {
    setupLog += "Data In: " + data + "\n";
    for(myKey in data){
	myData = data[myKey];
	setupLog += myKey + "," + myData + "; ";
    }
    setupLog += "\n"
    $.each(data, function (key, val)
	   {
	       setupLog += key + ',' + val + ';';
	       if(key == "PU")
	       {
		   setupLog += "keyPU1 ";
		   PanUpperSpeed = parseInt(val);
		   setupLog += "keyPU2 ";
		   $("#PanSpeedInput").val(calcControlDisplayVal(val, fPanResolution));
		   setupLog += "keyPU3 ";
	       }
	       else if(key == "TU")
	       {
		   TiltUpperSpeed = parseInt(val);
		   $("#TiltSpeedInput").val(calcControlDisplayVal(val, fTiltResolution));
	       }
	       if(key == "PL")
	       {
		   PanLowerSpeed = parseInt(val);
	       }
	       else if(key == "TL")
	       {
		   TiltLowerSpeed = parseInt(val);
	       }
	       else if(key == "PR")
	       {
		   fPanResolution = parseFloat(val);
	       }
	       else if(key == "TR")
	       {
		   fTiltResolution = parseFloat(val);
	       }
	       else if(key == "PX")
	       {
		   PanMax = val;
	       }
	       else if(key == "TX")
	       {
		   TiltMax = val;
	       }
	       else if(key == "PN")
	       {
		   PanMin = val;
	       }
	       else if(key == "TN")
	       {
		   TiltMin = val;
	       }
	       else if(key == "VM")
	       {
		   Model = val;
		   console.log("Model " + Model + "\n");
	       }
	       else if(key == "L")
	       {
		   LimitsMode = val; /* (F)actory, (U)ser, (D)iabled */
	       }
	   });
    setupLog += "\nDone with Parsing\n";
    $("#unitsdeg").attr("checked", true); 
    $("#controlvel").attr("checked", "checked");
    UpdateArrowVals();
    if(Model == 'E46' || Model == 'E47'){
	PanSwap = -1;
    }
}

/* Checks to see if the motion parameters have valid values.
 * If not, then tries to query them again
 */
function checkParameters(){
    if(retryCount > 5){
	return;
    }
    retryCount += 1;
    setTimeout(checkParameters, 1000*retryCount);

    var retryStr = "";
    if(PanUpperSpeed == 0){
	retryStr += "PU&";
    }
    if(TiltUpperSpeed == 0){
	retryStr += "TU&";
    }
    if(fPanResolution == 0){
	retryStr += "PR&";
    }
    if(fTiltResolution == 0){
	retryStr += "TR&";
    }
    if(LimitsMode == ''){
	retryStr += "L&";
    } else if(LimitsMode == 'U' || LimitsMode == 'F'){
	if(PanMax == 0){
	    retryStr += "PX&";
	}
	if(PanMin == 0) {
	    retryStr += "PN&";
	}
	if(TiltMax == 0){
	    retryStr += "TX&";
	}
	if(TiltMin == 0){
	    retryStr += "TN&";
	}
    }
    if(Model.length == 0){
	retryStr += "VM&";
    }

    if(retryStr.length > 0){
	retryStr += "V";
	setupLog += "Trying " + retryStr + " again\n";
	PTCmd(retryStr,
              parseParameters,
	  function () {
	      setupLog += "Retry parameter Error at " + retryCount + "\n";
	      showError();
	  }
	);
    } else {
	setupLog += "Parameters Good in " + retryCount + " tries.\n";
	retryCount = 10;  // Set high so we don't try agian.
    }
}



function UpdatePosSpeed()
{
    if(bMonitorWaiting == false)
    {
	bMonitorWaiting=true;
	var cmd = '' ;
	if(bGyroPresent)
	    cmd = 'SS&';
	cmd += PP+'&'+TP+'&'+PD+'&'+TD+'&C'
	$.ajax({data: cmd,
		success: UpdatePosSpeedFromJson,
		complete: function() { bMonitorWaiting=false;},
		error: function() { showError(); }
	       });
    }
}

function UpdateArrowVals()
{
    var smallspeed = 0;
    var bigspeed = 0;
    if(PanUpperSpeed < TiltUpperSpeed)
    {
	bigspeed = Math.round(PanUpperSpeed*0.75);
    }
    else
    {
	bigspeed = Math.round(TiltUpperSpeed*.75);
    }
    smallspeed = Math.round(bigspeed/10);
    
    if(smallspeed < PanLowerSpeed) {
	smallspeed = PanLowerSpeed;
    }
    if(smallspeed < TiltLowerSpeed) {
	smallspeed = TiltLowerSpeed;
    }
    
    if(Model == 'E46' || Model == 'E47') {
	PanSwap = -1;
    }
    
    var resolution = fPanResolution;
    if(fPanResolution > fTiltResolution)
    {
	resolution = fTiltResolution;
    }
    
    if(bUseDegrees)
    {
	$("#bigspeed").val(bigspeed*resolution);
	$("#smallspeed").val(smallspeed*resolution);
	$("#bigstep").val(100*resolution);
	$("#smallstep").val(10*resolution);
    }
    else
    {
	$("#bigspeed").val(bigspeed);
	$("#smallspeed").val(smallspeed);
	$("#bigstep").val(100);
	$("#smallstep").val(10);
    }

}

function calcControlDisplayVal(val, conv)
{
    setupLog += "cCDV1 ";
    if(bUseDegrees)
    {
	setupLog += "cCDV2 ";
/*	console.log("calcControlDisplayVal Degrees " + val + " " + conv); */
	setupLog += "cCDV3 ";
	var fVal = parseInt(val) * conv;
	setupLog += "cCDV4 ";
	return fVal.toFixed(3);
    }
    else
    {
	setupLog += "cCDV5 ";
	return parseInt(val);
    }
}


function SetDataUnits(handle, val, conv)
{
    /* don't update status if it's disabled */
    if(!bMonitorUpdating)
	return;
    if(bUseDegrees)
    {
	var fVal = parseInt(val) * conv;
	handle.html(fVal.toFixed(3));
    }
    else
    {
	handle.html(val);
    }

}

}

function UpdatePosSpeedFromJson(data)
{
    $.each(data, function(key,val) 
	   {
	       
	       switch(key)
	       {
	       case PP:
		   SetDataUnits($("#panpos"), val, fPanResolution);
		   break;
	       case TP:
		   SetDataUnits($("#tiltpos"), val, fTiltResolution);
		   break;
	       case PD:
		   iLastPanSpeed = val;
		   SetDataUnits($("#panspeed"), val, fPanResolution);
		   break;
	       case TD:
		   iLastTiltSpeed = val;
		   SetDataUnits($("#tiltspeed"), val, fTiltResolution);
		   break;
	       case 'C':
		   eLastMode = val;
		   break;
	       case 'SS':
		   UpdateStabState(val);
		   break;
	       default:
		   break;

	       }
	   });
}

function WaitForHome(iTries)
{

    if(iTries > 100)
	bWaitForHome = false;


    var pp = "";
    var tp = "";
    
    $.ajax( {
	data: PP+'&'+TP,
	
	success: function(data)
	{
	    $.each(data, function(key,val)
		   {
		       if(key == PP)
			   pp = val;
		       if(key == TP)
			   tp = val;
		   });
	    
	    if(pp == '0' && tp == '0')
		bWaitForHome = false;
	},
	complete: function(xmlReq, textRes)
	{
	    if(bWaitForHome == false)
	    {
		$.post("/API/PTCmd", 
		       "C=V",
		       GenericJson,
		       "json");
	    }
	    else
	    {
		iTries++;
		var cmd = "WaitForHome(" + iTries + ");";
		setTimeout(cmd, 250);
	    }
	    
	},
	error: function() {
	    showError();
	}
    });
    
}

function HandleStabState(data)
{
    $.each(data, function(key, val)
	   {
	       switch(key)
	       {
	       case 'SS':
		   UpdateStabState(val);
	       }

}

function UpdateStabState(val)
{
    var status = SSToString(val);
    switch(val)
    {
    case '1': //error
    case '5':
	$("#stabstart").val("Start");
	$("#stabstatus").html(status);
	bGyroPresent = true;
	break;
    case '2': //running
	$("#stabstart").val("Stop");
	$("#stabstatus").html(status);
	bGyroPresent = true;
	break;
    case '3': //calib
	$("#stabstart").val("N/A");
	$("#stabstatus").html(status);
	bGyroPresent = true;
	break;
    case '0': //not present
    case '4': //error
    case '6': //busy
	$("#stabstart").val("N/A");
	$("#stabstatus").html(status);
	bGyroPresent = false;
	break;
    }

}


function MouseDownFunc(id)
{
    var mode;
    if(bWaitForHome)
	return;
    
    
    
    if( $("input[name='controltype']:checked").val() == 'speed')
    {
	mode = 1;
	// get the speeds
	UpdatePosSpeed();
	bArrowDown = true;
    }
    else
    {
	mode = 0;
	bArrowDown = false;
    }
    
    errorLog += mode;
    var iBigStep = 0;
    var iSmallStep = 0;
    if(mode == 1)
    {
	iBigStep = $("#bigspeed").val();
	iSmallStep = $("#smallspeed").val();
    } 
    else
    {
	iBigStep = $("#bigstep").val();
	iSmallStep = $("#smallstep").val();	
    }

	}

function boundSpeed(speed, obj, max, min)
{
    sgn = sign(speed);
    if(Math.abs(speed) > max)
    {
	console.log("boundSpeed error: min " + parseInt(min) + ", max " + parseInt(max) + ", speed " + parseInt(speed));
	speed = sgn * max;
	showSpeedError(obj);
    }
    if(Math.abs(speed) < min)
    {
	console.log("boundSpeed error: min " + parseInt(min) + ", max " + parseInt(max) + ", speed " + parseInt(speed));
	speed = sgn * min;
	showSpeedError(obj);
    }
    return speed;
}

function showSpeedError(obj)
{
    showStep();
    obj.removeClass("whitebackground");
    obj.addClass("yellowbackground");
}

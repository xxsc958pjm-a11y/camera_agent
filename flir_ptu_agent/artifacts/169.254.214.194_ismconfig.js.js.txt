function SetBias(strCmd, fVal, pObject)
{
    var fRadBias = fVal ;
    var   strCmd = strCmd + "=" + fRadBias;
    if(biasWaiting == false)
    {
	biasWaiting = true;
	$.ajax({url: "/API/PTCmd", 
		type: "POST",
		data: strCmd, 
		success: function (data)
		{
		    pObject.val(fVal);
		    biasWaiting = false;
		}
	       });
    }
}

function SendEnabledControl()
{
    //$("#ControlForm tr input").css("border", "2px dotted blue");

    strCmd = '';
    $(".controldata").each( 
	function(index)
	{
	    strCmd  = strCmd + $(this).attr('name') + '=' + $(this).val() + '&';

	});
    //alert(strCmd);
    $.post("/API/PTCmd", 
	   strCmd,
	   GenericJson,
	   "json");
}

function UpdateDataFromJson(data)
{
    $.each(data, function(key, val) {
	
	switch(key)
	{
	case 'V':
	    $("#firmware").html(val);
	    break;
	case 'VM':
	    $("#model").html(val);
	    break;
	case 'SS':
	    $("#status").html(SSToString(val));
	    break;
	}
    });
}

//This function is here so it waits for the calib task to start before querying status
function CheckCalibrationDoneWrapper()
{
    restartCnt = 5;
    setTimeout(CheckCalibrationDone, 500);
}

function CheckCalibrationDone()
{
    var status;
    $("#status").html("Calibrating");
    PTCmd("SS",
	  function (data)
	  {
	      $.each(data, function (key,val)
		     {
			 switch(key)
			 {
			 case 'SS':
			     status = SSToString(val);
			     $("#status").html(status);
			     if(status == "Calibration<BR>Failed")
			     {
				 alert("Calibration Failed");
				 hideLoading();
			     }
			     else if(status == "Ready")
			     {
				 alert("Calibration Successful");
				 UpdateData();
				 hideLoading();
			     }
			     else if(status == "Calibrating")
			     {
				 setTimeout(CheckCalibrationDone, 5000);
			     } else {
				 if(restartCnt == 0){
				     alert(status);
				     UpdateData();
				     hideLoading();
				 } else {
				     restartCnt--;
				     setTimeout(CheckCalibrationDone, 5000);
				 }
			     }
			     
			 }
		     });
	  },  function(){
	      restartCnt--;
	      setTimeout(CheckCalibrationDone, 5000);
	  }, '');
}

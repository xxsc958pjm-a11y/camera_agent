function UpdateData()
{

    $.post("/API/PTCmd", 
	   "V&VM",
	   UpdateDataFromJson,
	   "json");
    $.post("/API/PTCmd",
	   "GS&GPT",
	   UpdateDataFromJson,
	   "json");
    $.post("/API/PTCmd",
	   "GL&GO&GA",
	   UpdateDataFromJson,
	   "json");
    $.post("/API/PTCmd",
	   "GR&GP&GY&GCP",
	   UpdateDataFromJson,
	   "json");
    $.post("/API/PTCmd",
	   "GCQ",
	   UpdateDataFromJson,
	   "json");
    FetchLandmarks();
    
}


function UpdateDataFromJson(data)
{
    $.each(data, function(key, val) {
	
     	switch(key)
	{
	case 'GA':
	    $("#GA").val(val);
	    break;
	case 'GCP':
	    $("#GCP").val(val);
	    break;
	case 'GCQ':
	    $("#GCQ").html(val);
	    break;
	case 'GL':
	    $("#GL").val(val);
	    break;
	case 'GO':
	    $("#GO").val(val);
	    break;
	case 'GP':
	    $("#GP").val(val);
	    break;
	case 'GPT':
	    if(val == "0")
	    {
		$("#GPM0").attr('checked', true);
		$("#GPM1").attr('checked', false);
	    }
	    else if(val == "1")
	    {
		$("#GPM0").attr('checked', false);
		$("#GPM1").attr('checked', true);
	    }
	    break;
	case 'GR':
	    $("#GR").val(val);
	    break;
	case 'GS':
	    $("#GS").html(GPMStatusStr(val));
	    break;
	case 'GY':
	    $("#GY").val(val);
	    break;
	case 'V':
	    $("#firmware").html(val);
	    break;
	case 'VM':
	    $("#model").html(val);
	    break;
	}
	
    });
}
